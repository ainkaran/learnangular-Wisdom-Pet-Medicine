# Git: configurations < Register your username and email with local repositories
$ git config --global user.name "King Kong"
$ git config --global user.email "king-kong@gmail.com"
$ git config --global color.ui true
$ git config --list

# Git: starting a repository < Initialize a local repository
$ git init
Initialized empty Git repository in /home/dell/new-folder/.git/
$ git status

# Git: staging files < Add and remove files to and from the staging area
$ git add my_new_file.txt
$ git add my-file.ts another-file.js new_file.rb
$ git add .
$ git add --all
$ git add -A
$ git rm --cached my-file.ts
$ rm 'my_file.ts'
$ git reset another-file.js

# Git: commiting to a repository < Commit changes to the repository and Undo commits
$ git commit -m "Add three files"
$ git commit -a -m "Do something once more"
$ git reset --soft HEAD^ 
$ git add file-i-forgot-to-add.html
$ git commit --amend -m "Add the remaining file"
